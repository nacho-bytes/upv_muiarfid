
DOMINIO EJEMPLO

(define (domain EJEMPLO)
(:requirements :strips :typing :equality :negative-preconditions)

(:types   place locatable - object
	  house office - place
          person box vehicle - locatable
          car truck van - vehicle)

(:predicates (at ?x - locatable ?p - place)
             (in ?o - (either person box) ?v - vehicle))
